## Install

```bash
npm install --save react-file-manager-rfm
```

## Usage

```jsx
import React, { Component } from 'react'

import RFM from 'react-file-manager-rfm'
import 'react-file-manager-rfm/dist/index.css'

class Example extends Component {
  render() {
    return <RFM />
  }
}
```
